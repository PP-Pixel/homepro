#include <Servo.h>

#include <Wire.h>

#include <VL53L0X.h>



VL53L0X sensor;

Servo myservo;



void setup()

{

        // サーボ設定

        myservo.attach(9);

        myservo.write(90);



        // LEDピンの設定

        pinMode(8, OUTPUT);

        digitalWrite(8, LOW);



        // 距離センサーの初期設定

        Serial.begin(9600);

        Wire.begin();



        sensor.setTimeout(500);

        if (!sensor.init())

        {

                Serial.println("Failed to detect and initialize sensor!");

                while (1)

                {}

        }

}



// スイッチをOFFにする

void SwOff()

{

        myservo.write(90); // 元の角度に戻す

        delay(500);

        myservo.write(140); // ここで動かす角度を調整

        delay(500);

        .write(90); // 元の角度に戻す

}



// スイッチをONにする

void SwOn()

{

        myservo.write(90); // 元の角度に戻す

        delay(500);

        myservo.write(60); // ここで動かす角度を調整

        delay(500);

        myservo.write(90); // 元の角度に戻す

}



// 距離を測定する

bool CheckDistance()

{

        int distance;

        distance = sensor.readRangeSingleMillimeters();

        Serial.println(distance);

        if ((50 < distance) && (distance < 110))

        { // 5cm以上11cm未満で動かす

                return true;

        }

        return false;

}



void loop()

{

    

        bool sw_onoff = false;

        int count = 0;

    

        // はじめに電源OFFする

        SwOff();

    

        // ずっと回るループ

        while (1)

        {

        

                // 距離を測定し、設定範囲だったらカウントアップ、範囲外なら0にする

                if (CheckDistance())

                {

                        count++;

                        digitalWrite(8, HIGH); // LEDを点灯

                }

                else

                {

                        count = 0;

                        digitalWrite(8, LOW); // LEDを消灯

                }

                

                // 10回連続で距離が設定範囲だったとき

                if (count == 10)

                {

                        // 現在スイッチOFFの場合

                        if (sw_onoff == false)

                        {

                                sw_onoff = true; // スイッチON状態を記憶

                                SwOn();          // サーボモータをONの制御

                        }

                        // 現在スイッチONの場合

                        else

                        {

                                sw_onoff = false; // スイッチOFF状態を記憶

                                SwOff();          // サーボモータをOFFの制御

                        }

                        

                        // 動作後は、カウントを0にする

                        count = 0;

                }

                

            // 0.1秒待つ（ループが回る速さを制限する）

            delay(100);

        }

}
